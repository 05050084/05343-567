en=imread('cameraman.tif');
[x,y]=meshgrid(1:256,1:256);
s=1+sin(x/3+y/5);
ep=(double(en)/128+s)/4;

figure,imshow(ep);
cf=fftshift(fft2(ep));
figure,fftshow(cf);


z=sqrt((x-129).^2+(y-129).^2);
br=(z<15|z>20);
tbr=cf.*br;
figure,fftshow(tbr,'log');
cfli=ifft2(cf);
figure,fftshow(cfli,'abs');

cf(119:121,:)=0;
cf(135:137,:)=0;
cf(:,113:115)=0;
cf(:,141:143)=0;
figure,fftshow(cf,'log');
cfli=ifft2(cf);
figure,fftshow(cfli,'abs');
